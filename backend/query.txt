SELECT 
    P.name AS player_name,
    P.nationality,
    P.date_of_birth,
    P.current_club,
    P.goals,
    P.assists,
    Pos.position_name
FROM 
    UserPlayers UP
JOIN 
    Players P ON UP.player_id = P.player_id
JOIN 
    Positions Pos ON UP.position_id = Pos.position_id
WHERE 
    UP.user_id = 1; -- Replace with the user_id of interest
